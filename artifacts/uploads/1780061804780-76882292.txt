-- Sistema de Aimbot
local Camera = workspace.CurrentCamera
local LocalPlayer = Players.LocalPlayer
local fovCircle = nil
local lockedTarget = nil
local FOVRadius = 200

local function criarFOVCircle()
    if fovCircle then
        fovCircle:Remove()
    end
    
    fovCircle = Drawing.new("Circle")
    fovCircle.Radius = FOVRadius
    fovCircle.Color = Color3.fromRGB(0, 255, 0)
    fovCircle.Thickness = 2
    fovCircle.Filled = false
    fovCircle.Visible = aimbotEnabled
    fovCircle.Position = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
end

local function getBestTarget()
    local center = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
    local closest = nil
    local closestDist = math.huge

    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local hum = player.Character:FindFirstChild("Humanoid")
            if hum and hum.Health > 0 then
                local head = player.Character:FindFirstChild("Cabesa") or player.Character:FindFirstChild("Head")
                if head then
                    local screenPos, onScreen = Camera:WorldToViewportPoint(head.Position)
                    if onScreen then
                        local rayParams = RaycastParams.new()
                        rayParams.FilterDescendantsInstances = {LocalPlayer.Character}
                        rayParams.FilterType = Enum.RaycastFilterType.Blacklist

                        local rayResult = workspace:Raycast(Camera.CFrame.Position, (head.Position - Camera.CFrame.Position).Unit * 1000, rayParams)
                        if rayResult and rayResult.Instance:IsDescendantOf(player.Character) then
                            local dist = (Vector2.new(screenPos.X, screenPos.Y) - center).Magnitude
                            if dist < closestDist and dist <= FOVRadius then
                                closest = head
                                closestDist = dist
                            end
                        end
                    end
                end
            end
        end
    end

    return closest
end

local function toggleAimbot(ativado)
    aimbotEnabled = ativado
    toggleStates["Aimbot (FOV)"] = ativado
    
    if ativado then
        criarFOVCircle()
        
        aimbotConnection = RunService.RenderStepped:Connect(function(delta)
            if not aimbotEnabled then
                if fovCircle then
                    fovCircle.Visible = false
                end
                return
            end

            -- Mantém o círculo sempre no centro da tela
            if fovCircle then
                fovCircle.Position = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
                fovCircle.Radius = FOVRadius
                fovCircle.Visible = true
            end

            local target = getBestTarget()
            if target then
                Camera.CFrame = CFrame.new(Camera.CFrame.Position, target.Position)
                lockedTarget = target
            end
        end)
        
        print("🎯 Aimbot ativado")
    else
        if aimbotConnection then
            aimbotConnection:Disconnect()
            aimbotConnection = nil
        end
        
        if fovCircle then
            fovCircle.Visible = false
        end
        
        lockedTarget = nil
        print("🎯 Aimbot desativado")
    end
end

-- Sistema Noclip
local function setCollisions(collide)
    if Player.Character then
        for _, part in ipairs(Player.Character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = collide
            end
        end
    end
end

local function toggleNoclip(ativado)
    noclipAtivado = ativado
    toggleStates["Noclip Automático"] = ativado
    
    if ativado then
        if noclipConexao then
            noclipConexao:Disconnect()
        end
        
        noclipConexao = RunService.RenderStepped:Connect(function()
            setCollisions(false)
        end)
        
        print("✅ Noclip ativado")
    else
        if noclipConexao then
            noclipConexao:Disconnect()
            noclipConexao = nil
        end
        
        setCollisions(true)
        print("✅ Noclip desativado")
    end
end

-- Sistema Anti-Algemas
local HandcuffEvent = ReplicatedStorage:FindFirstChild("HandcuffPlayer")

local function toggleAntiAlgemas(ativado)
    antiAlgemasAtivado = ativado
    toggleStates["Anti-Algemas"] = ativado
    
    if ativado then
        if HandcuffEvent then
            antiAlgemasConnection = HandcuffEvent.OnClientEvent:Connect(function(officer, victim)
                -- Se somos a vítima, prevenir o efeito de algema
                if victim == Player then
                    -- Restaurar movimentação imediatamente
                    local humanoid = Player.Character and Player.Character:FindFirstChild("Humanoid")
                    if humanoid then
                        humanoid.WalkSpeed = 16  -- Velocidade normal
                        humanoid.JumpPower = 50  -- Pulo normal
                    end
                    
                    -- Opcional: Notificar que foi protegido
                    print("🛡️ Anti-Algemas: Protegido contra algema de " .. officer.Name)
                end
            end)
        end
        
        -- Monitorar também mudanças no WalkSpeed (backup)
        if Player.Character then
            local humanoid = Player.Character:FindFirstChild("Humanoid")
            if humanoid then
                humanoid:GetPropertyChangedSignal("WalkSpeed"):Connect(function()
                    if antiAlgemasAtivado and humanoid.WalkSpeed == 0 then
                        humanoid.WalkSpeed = 16
                    end
                end)
            end
        end
        
        Player.CharacterAdded:Connect(function(character)
            local humanoid = character:WaitForChild("Humanoid")
            humanoid:GetPropertyChangedSignal("WalkSpeed"):Connect(function()
                if antiAlgemasAtivado and humanoid.WalkSpeed == 0 then
                    humanoid.WalkSpeed = 16
                end
            end)
        end)
        
        print("🛡️ Anti-Algemas ativado")
    else
        if antiAlgemasConnection then
            antiAlgemasConnection:Disconnect()
            antiAlgemasConnection = nil
        end
        
        print("🛡️ Anti-Algemas desativado")
    end
end

    local label = Instance.new("TextLabel")        
    label.Parent = toggleFrame        
    label.BackgroundTransparency = 1        
    label.Position = UDim2.new(0.03, 0, 0, 0)        
    label.Size = UDim2.new(0.7, 0, 1, 0)        
    label.Font = Enum.Font.Gotham        
    label.Text = nome        
    label.TextColor3 = Color3.fromRGB(0, 0, 0)        
    label.TextSize = 14        
    label.TextXAlignment = Enum.TextXAlignment.Left        
    label.ZIndex = 7
    
    local btn = Instance.new("TextButton")        
    btn.Parent = toggleFrame        
    btn.BackgroundColor3 = Color3.fromRGB(200, 60, 60)        
    btn.Position = UDim2.new(0.82, 0, 0.15, 0)        
    btn.Size = UDim2.new(0, 45 * scaleMultiplier, 0, 22 * scaleMultiplier)      
    btn.Font = Enum.Font.GothamBold        
    btn.Text = "OFF"        
    btn.TextColor3 = Color3.fromRGB(255, 255, 255)        
    btn.TextSize = 11        
    btn.ZIndex = 7
    
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 10)
    
    -- Verificar estado salvo
    local estadoSalvo = toggleStates[nome] or false
    
    if estadoSalvo then
        btn.BackgroundColor3 = Color3.fromRGB(0, 180, 0)
        btn.Text = "ON"
        -- Ativar o sistema se estiver salvo como ativo
        if callback then 
            task.spawn(function()
                callback(true)
            end)
        end
    end
    
    btn.MouseButton1Click:Connect(function()
        if not menuAtivo then return end
        local ativado = not estadoSalvo
        estadoSalvo = ativado
        toggleStates[nome] = ativado
        
        btn.BackgroundColor3 = ativado and Color3.fromRGB(0, 180, 0) or Color3.fromRGB(200, 60, 60)        
        btn.Text = ativado and "ON" or "OFF"        
        
        if callback then callback(ativado) end        
    end)        
    
    table.insert(opcoes, {frame = toggleFrame, nome = nome, aba = abaAtual, tipo = "toggle", elemento = btn})
    
    return btn
end

    sliderBack.Position = UDim2.new(0.05, 0, 0.5, 0)        
    sliderBack.Size = UDim2.new(0.9, 0, 0, 6)      
    sliderBack.ZIndex = 7
    Instance.new("UICorner", sliderBack).CornerRadius = UDim.new(1, 0)        
    
    local sliderFill = Instance.new("Frame")        
    sliderFill.Parent = sliderBack        
    sliderFill.BackgroundColor3 = Color3.fromRGB(0, 100, 200)        
    sliderFill.Size = UDim2.new(0, 0, 1, 0)        
    sliderFill.ZIndex = 7
    Instance.new("UICorner", sliderFill).CornerRadius = UDim.new(1, 0)        
    
    local sliderButton = Instance.new("TextButton")        
    sliderButton.Parent = sliderBack        
    sliderButton.BackgroundColor3 = Color3.fromRGB(255, 255, 255)        
    sliderButton.Position = UDim2.new(0, -6, 0.5, -6)      
    sliderButton.Size = UDim2.new(0, 12 * scaleMultiplier, 0, 12 * scaleMultiplier)      
    sliderButton.Text = ""        
    sliderButton.ZIndex = 8
    Instance.new("UICorner", sliderButton).CornerRadius = UDim.new(1, 0)        
    
    -- Configurar posição inicial
    local relativePos = (valorInicial - min) / (max - min)
    sliderFill.Size = UDim2.new(relativePos, 0, 1, 0)        
    sliderButton.Position = UDim2.new(relativePos, -6, 0.5, -6)
    
    local dragging = false        
    sliderButton.MouseButton1Down:Connect(function()
        if not menuAtivo then return end
        dragging = true        
    end)        
    
    UserInputService.InputEnded:Connect(function(input)        
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then        
            dragging = false        
        end        
    end)        
    
    UserInputService.InputChanged:Connect(function(input)        
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then        
            local mousePos = UserInputService:GetMouseLocation().X        
            local sliderPos = sliderBack.AbsolutePosition.X        
            local sliderSize = sliderBack.AbsoluteSize.X        
            local relativePos = math.clamp((mousePos - sliderPos) / sliderSize, 0, 1)        
            local value = math.floor(min + (max - min) * relativePos)        
            sliderFill.Size = UDim2.new(relativePos, 0, 1, 0)        
            sliderButton.Position = UDim2.new(relativePos, -6, 0.5, -6)        
            valueLabel.Text = tostring(value)        
            if callback then callback(value) end        
        end        
    end)        
    
    table.insert(opcoes, {frame = sliderFrame, nome = nome, aba = abaAtual, tipo = "slider"})
end

-- Função para tornar arrastável
local function tornarArrastavel(frame)
    local dragging = false
    local dragStart, startPos
    
    frame.InputBegan:Connect(function(input)  
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then  
            dragging = true  
            dragStart = input.Position  
            startPos = mainContainer.Position  
        end  
    end)  
    
    frame.InputEnded:Connect(function(input)  
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then  
            dragging = false  
        end  
    end)  
    
    UserInputService.InputChanged:Connect(function(input)  
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then  
            local delta = input.Position - dragStart  
            mainContainer.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)  
        end  
    end)
end

-- Aplicar arrastável
tornarArrastavel(dragFrame)

-- Função de animação
local function animacaoMexer(frame)
    if not menuAtivo then return end
    local originalPosition = frame.Position
    frame:TweenPosition(
        originalPosition + UDim2.new(0, 2, 0, 2),
        Enum.EasingDirection.Out,
        Enum.EasingStyle.Quad,
        0.05,
        true,
        function()
            frame:TweenPosition(
                originalPosition,
                Enum.EasingDirection.Out,
                Enum.EasingStyle.Back,
                0.1,
                true
            )
        end
    )
end

-- Função para trocar de aba
local function trocarAba(nomeAba)
    if not menuAtivo then return end
    abaAtual = nomeAba
    nomeaba.Text = nomeAba
    pesquisar.Text = ""

    -- Limpar conteúdo anterior
    for _, child in pairs(ScrollFrame:GetChildren()) do    
        if child:IsA("Frame") then child:Destroy() end    
    end    
    
    opcoes = {}        
    
    if nomeAba == "Inicio" then        
        -- Conteúdo da aba Inicio
        criarToggle("Fly PC (E)", function(ativado)
            print("Fly PC: " .. (ativado and "Ativado" or "Desativado"))
        end)
        
    elseif nomeAba == "Combate" then
        -- Apenas os 3 toggles principais
        criarToggle("Noclip Automático", function(ativado)
            toggleNoclip(ativado)
        end)
        
        criarToggle("Anti-Algemas", function(ativado)
            toggleAntiAlgemas(ativado)
        end)
        
    elseif nomeAba == "Aimbots" then
        -- Aimbot na aba Aimbots
        criarToggle("Aimbot (FOV)", function(ativado)
            toggleAimbot(ativado)
        end)
        
        -- Slider para FOV do Aimbot
        criarSlider("FOV Aimbot", 50, 500, FOVRadius, function(valor)
            FOVRadius = valor
            if fovCircle then
                fovCircle.Radius = FOVRadius
            end
            print("FOV ajustado para: " .. valor)
        end)
        
    else
        -- Para outras abas, manter apenas um toggle de exemplo
        criarToggle("Exemplo " .. nomeAba, function(ativado)
            print(nomeAba .. ": " .. (ativado and "Ativado" or "Desativado"))
        end)
    end    
    
    task.wait(0.1)    
    ScrollFrame.CanvasSize = UDim2.new(0, 0, 0, UIListLayout.AbsoluteContentSize.Y + 20)
end

local nomesAba = {"Inicio", "Combate", "Esp's", "Aimbots", "+Revistar", "Teleportes", "Veículo", "Farm", "Configurações", "Extras"}
