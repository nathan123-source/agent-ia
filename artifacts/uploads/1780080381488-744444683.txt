-- Hitbox
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local headModEnabled = false
local headSize = 15
local friendlyList = {}
local hitboxLoop = nil

local function updateHitbox()
    if headModEnabled then
        for _, plr in pairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character ~= nil and not friendlyList[plr.Name] then
                for _, part in pairs(plr.Character:GetChildren()) do
                    if part:IsA("BasePart") and (part.Name:lower():find("head") or part.Name:lower():find("cabeca") or part.Name:lower():find("cabesa")) then
                        part.Size = Vector3.new(headSize, headSize, headSize)
                        part.Transparency = 0.7
                        part.CanCollide = false
                        part.Massless = true
                    end
                end
            elseif friendlyList[plr.Name] and plr.Character ~= nil then
                for _, part in pairs(plr.Character:GetChildren()) do
                    if part:IsA("BasePart") and (part.Name:lower():find("head") or part.Name:lower():find("cabeca") or part.Name:lower():find("cabesa")) then
                        part.Size = Vector3.new(2, 1, 1)
                        part.Transparency = 0
                        part.CanCollide = true
                        part.Massless = false
                    end
                end
            end
        end
    else
        for _, plr in pairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character ~= nil then
                for _, part in pairs(plr.Character:GetChildren()) do
                    if part:IsA("BasePart") and (part.Name:lower():find("head") or part.Name:lower():find("cabeca") or part.Name:lower():find("cabesa")) then
                        part.Size = Vector3.new(2, 1, 1)
                        part.Transparency = 0
                        part.CanCollide = true
                        part.Massless = false
                    end
                end
            end
        end
    end
end

local function startHitboxLoop()
    if hitboxLoop then return end
    
    hitboxLoop = game:GetService("RunService").Heartbeat:Connect(function()
        updateHitbox()
    end)
end

local function stopHitboxLoop()
    if hitboxLoop then
        hitboxLoop:Disconnect()
        hitboxLoop = nil
    end
end

-- ESP Highlight
local espAtivo = false
local espStorage = nil
local espConnections = {}

local function criarESPStorage()
    if not espStorage then
        espStorage = Instance.new("Folder")
        espStorage.Parent = game:GetService("CoreGui")
        espStorage.Name = "Highlight_Storage"
    end
    return espStorage
end

local function limparESP()
    if espStorage then
        espStorage:Destroy()
        espStorage = nil
    end
    
    for player, connection in pairs(espConnections) do
        if connection then
            connection:Disconnect()
        end
    end
    espConnections = {}
end

local function aplicarHighlight(plr)
    if not espAtivo or not plr then return end
    
    local Highlight = Instance.new("Highlight")
    Highlight.Name = plr.Name
    Highlight.FillColor = Color3.fromRGB(175, 25, 255)
    Highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    Highlight.FillTransparency = 0.5
    Highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
    Highlight.OutlineTransparency = 0
    Highlight.Parent = criarESPStorage()
    
    local plrchar = plr.Character
    if plrchar then
        Highlight.Adornee = plrchar
    end

    espConnections[plr] = plr.CharacterAdded:Connect(function(char)
        if espAtivo and Highlight then
            Highlight.Adornee = char
        end
    end)
end

local function iniciarESPHighlight()
    if espAtivo then return end
    
    espAtivo = true
    limparESP()
    
    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= Players.LocalPlayer then
            aplicarHighlight(player)
        end
    end
    
    espConnections.playerAdded = Players.PlayerAdded:Connect(function(player)
        aplicarHighlight(player)
    end)
    
    espConnections.playerRemoving = Players.PlayerRemoving:Connect(function(plr)
        if espStorage and espStorage:FindFirstChild(plr.Name) then
            espStorage[plr.Name]:Destroy()
        end
        if espConnections[plr] then
            espConnections[plr]:Disconnect()
            espConnections[plr] = nil
        end
    end)
    
    DiscordLib:Notification("ESP", "ESP Highlight ativado!", "ok")
end

local function pararESPHighlight()
    if not espAtivo then return end
    
    espAtivo = false
    limparESP()
    DiscordLib:Notification("ESP", "ESP Highlight desativado!", "fail")
end

VisualChannel:Toggle("ESP Highlight", false, function(state)
    if state then
        iniciarESPHighlight()
    else
        pararESPHighlight()
    end
end)

-- AUTO FARM GARI
local AutoFarmGari = false
local farmGariConnection = nil
local lastSwitchTimeGari = 0
local currentTargetIndexGari = 1

local pontosGari = {
    CFrame.new(-1548.10632, 4.02870989, -405.713684),
    CFrame.new(-1546.32886, 4.02870941, -382.746277)
}

local lastProximityCheckGari = 0
local proximityCheckIntervalGari = 2

local function handleProximityPromptsGari(hrp)
    local now = tick()
    if now - lastProximityCheckGari < proximityCheckIntervalGari then return end
    lastProximityCheckGari = now
    
    for _, prompt in ipairs(workspace:GetDescendants()) do
        if not AutoFarmGari then break end
        if prompt:IsA("ProximityPrompt") and prompt.Enabled then
            local part = prompt.Parent
            if part and part:IsA("BasePart") then
                local distance = (part.Position - hrp.Position).Magnitude
                if distance <= 15 then
                    fireproximityprompt(prompt)
                    task.wait(0.2)
                    break
                end
            end
        end
    end
end

local function StartAutoFarmGari()
    local RunService = game:GetService("RunService")
    
    if farmGariConnection then
        farmGariConnection:Disconnect()
        farmGariConnection = nil
    end
    
    farmGariConnection = RunService.Heartbeat:Connect(function()
        if not AutoFarmGari then return end
        
        local character = LocalPlayer.Character
        if not character then return end
        
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        local hrp = character:FindFirstChild("HumanoidRootPart")
        
        if not humanoid or humanoid.Health <= 0 or not hrp then return end
        
        local now = tick()
        if now - lastSwitchTimeGari >= 5 then
            lastSwitchTimeGari = now
            currentTargetIndexGari = currentTargetIndexGari == 1 and 2 or 1
            hrp.CFrame = pontosGari[currentTargetIndexGari]
        end
        
        handleProximityPromptsGari(hrp)
    end)
end

-- Função para encontrar o jogador mais próximo dentro do FOV
local function GetClosestPlayer()
    if not LocalPlayer or not LocalPlayer.Character then return nil end
    
    local MaximumDistance = aimbotConfig.FOVRadius
    local Target = nil
    
    for _, Player in pairs(Players:GetPlayers()) do
        if Player ~= LocalPlayer and Player.Character then
            -- Verificar lista de amigos
            if aimbotConfig.FriendlyList[Player.Name] then
                continue
            end
            
            local Character = Player.Character
            local Humanoid = Character:FindFirstChild("Humanoid")
            local HumanoidRootPart = Character:FindFirstChild("HumanoidRootPart")
            local TargetPart = Character:FindFirstChild(aimbotConfig.AimPart)
            
            -- Verificar se o jogador é válido
            if Humanoid and HumanoidRootPart and TargetPart then
                -- Verificação de kill check
                if aimbotConfig.KillCheck and Humanoid.Health <= 0 then
                    continue
                end
                
                -- Verificação de time
                if aimbotConfig.TeamCheck then
                    if LocalPlayer.Team and Player.Team and Player.Team == LocalPlayer.Team then
                        continue
                    end
                end
                
                -- Wall check
                if aimbotConfig.WallCheck then
                    local raycastParams = RaycastParams.new()
                    raycastParams.FilterDescendantsInstances = {LocalPlayer.Character}
                    raycastParams.FilterType = Enum.RaycastFilterType.Blacklist
                    raycastParams.IgnoreWater = true
                    
                    local raycastResult = workspace:Raycast(Camera.CFrame.Position, (TargetPart.Position - Camera.CFrame.Position).unit * 500, raycastParams)
                    if raycastResult and raycastResult.Instance and not raycastResult.Instance:IsDescendantOf(Player.Character) and (raycastResult.Position - TargetPart.Position).Magnitude > 2 then
                        continue
                    end
                end
                
                -- Converter posição para a tela
                local ScreenPoint, OnScreen = Camera:WorldToViewportPoint(TargetPart.Position)
                
                if OnScreen then
                    -- Calcular distância do centro da tela
                    local ScreenCenter = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
                    local VectorDistance = (ScreenCenter - Vector2.new(ScreenPoint.X, ScreenPoint.Y)).Magnitude
                    
                    -- Verificar se está dentro do FOV
                    if VectorDistance < MaximumDistance then
                        MaximumDistance = VectorDistance
                        Target = Player
                    end
                end
            end
        end
    end
    
    return Target
end

-- Loop principal do aimbot
local aimbotConnection = nil

local function iniciarAimbot()
    if aimbotConnection then return end
    
    aimbotConnection = RunService.RenderStepped:Connect(function()
        UpdateCircle()
        
        if aimbotConfig.AimLockActive and aimbotConfig.Enabled then
            local Target = GetClosestPlayer()
            
            if Target and Target.Character then
                local TargetPart = Target.Character:FindFirstChild(aimbotConfig.AimPart)
                
                if TargetPart then
                    local TargetCFrame = CFrame.new(Camera.CFrame.Position, TargetPart.Position)
                    
                    if aimbotConfig.Smoothness > 0 then
                        local alpha = math.clamp(aimbotConfig.Smoothness / 100, 0.01, 1)
                        Camera.CFrame = Camera.CFrame:Lerp(TargetCFrame, alpha)
                    else
                        Camera.CFrame = TargetCFrame
                    end
                end
            end
        end
    end)
end
