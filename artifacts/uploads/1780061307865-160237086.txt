local function CreateESP(player)
    local Box = Drawing.new("Square")
    Box.Color = Color3.fromRGB(255, 255, 255)
    Box.Thickness = 1
    Box.Filled = false

    local HealthBarOutline = Drawing.new("Line")
    HealthBarOutline.Thickness = 2
    HealthBarOutline.Color = Color3.fromRGB(0, 0, 0)

    local HealthBar = Drawing.new("Line")
    HealthBar.Thickness = 1.5
    HealthBar.Color = Color3.fromRGB(0, 255, 0)

    local connection
    connection = RunService.RenderStepped:Connect(function()
        if not espBoxEnabled then
            Box.Visible = false
            HealthBar.Visible = false
            HealthBarOutline.Visible = false
            return
        end
        
        if player.Character and player.Character:FindFirstChild("HumanoidRootPart") and player.Character:FindFirstChild("Humanoid") then
            local root = player.Character.HumanoidRootPart
            local humanoid = player.Character.Humanoid

            local pos, vis = Camera:WorldToViewportPoint(root.Position)
            if vis then
                local headPos = Camera:WorldToViewportPoint(root.Position + Vector3.new(0, 3, 0))
                local footPos = Camera:WorldToViewportPoint(root.Position - Vector3.new(0, 3, 0))

                local height = math.abs(headPos.Y - footPos.Y)
                local width = height / 1.4 -- 🔹 mais largo (de 1.7 → 1.4)

                Box.Size = Vector2.new(width, height * 0.95) -- 🔹 levemente mais baixo
                Box.Position = Vector2.new(pos.X - width / 2, pos.Y - (height * 0.95) / 2)
                Box.Visible = true

                
                local healthPercent = math.clamp(humanoid.Health / humanoid.MaxHealth, 0, 1)
                local y1 = Box.Position.Y
                local y2 = Box.Position.Y + Box.Size.Y

                local barX = Box.Position.X - 6
                local barY = y2
                local barHeight = (y2 - y1) * healthPercent

                -- Fundo preto
                HealthBarOutline.From = Vector2.new(barX, y1)
                HealthBarOutline.To = Vector2.new(barX, y2)
                HealthBarOutline.Visible = true

                -- Vida colorida (verde → vermelho)
                HealthBar.From = Vector2.new(barX, y2)
                HealthBar.To = Vector2.new(barX, y2 - barHeight)
                HealthBar.Color = Color3.fromRGB(255 * (1 - healthPercent), 255 * healthPercent, 0)
                HealthBar.Visible = true
            else
                Box.Visible = false
                HealthBar.Visible = false
                HealthBarOutline.Visible = false
            end
        else
            Box.Visible = false
            HealthBar.Visible = false
            HealthBarOutline.Visible = false
        end
    end)
    
    espBoxConnections[player] = connection
end

local function setupESPSystem()
    local function enableESP()
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= Players.LocalPlayer then
                CreateESP(p)
            end
        end
        
        Players.PlayerAdded:Connect(function(p)
            if p ~= Players.LocalPlayer then
                CreateESP(p)
            end
        end)
    end
    
    local function disableESP()
        for player, connection in pairs(espBoxConnections) do
            if connection then
                connection:Disconnect()
            end
        end
        espBoxConnections = {}
    end
    
    return {
        enable = enableESP,
        disable = disableESP
    }
end

local function setupHighlightSystem()

        Highlight.OutlineTransparency = 0
        Highlight.Parent = Storage
        local function updateAdornee(char)
            Highlight.Adornee = char
        end
        if plr.Character then 
            updateAdornee(plr.Character) 
        end
        highlightConnections[plr] = plr.CharacterAdded:Connect(updateAdornee)
    end
    local function enableHighlight()
        for _, p in pairs(Players:GetPlayers()) do 
            Highlight(p) 
        end
        Players.PlayerAdded:Connect(Highlight)
    end
    local function disableHighlight()
        local Storage = CoreGui:FindFirstChild("Highlight_Storage")
        if Storage then
            for _, child in pairs(Storage:GetChildren()) do 
                child:Destroy() 
            end
        end
        for _, conn in pairs(highlightConnections) do 
            conn:Disconnect() 
        end
        highlightConnections = {}
    end
    return {
        enable = enableHighlight,
        disable = disableHighlight
    }
end

local function getBestTarget()
    local center = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
    local closest = nil
    local closestDist = math.huge
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= Players.LocalPlayer and player.Character then
            local hum = player.Character:FindFirstChild("Humanoid")
            if killCheckEnabled then
                if not hum or hum.Health <= 0 then
                    continue
                end
            else
                if not hum then
                    continue
                end
            end
            local head = player.Character:FindFirstChild("Cabesa") or player.Character:FindFirstChild("Head")
            if head then
                local screenPos, onScreen = Camera:WorldToViewportPoint(head.Position)
                if onScreen then
                    if wallCheckEnabled then
                        local rayParams = RaycastParams.new()
                        rayParams.FilterDescendantsInstances = {Players.LocalPlayer.Character}
                        rayParams.FilterType = Enum.RaycastFilterType.Blacklist
                        local rayResult = workspace:Raycast(Camera.CFrame.Position, (head.Position - Camera.CFrame.Position).Unit * 1000, rayParams)
                        if rayResult then
                            if rayResult.Instance:IsDescendantOf(player.Character) then
                                local dist = (Vector2.new(screenPos.X, screenPos.Y) - center).Magnitude
                                if dist < closestDist and dist <= FOVRadius then
                                    closest = head
                                    closestDist = dist
                                end
                            else
                                continue
                            end
                        else
                            continue
                        end
                    else
                        local dist = (Vector2.new(screenPos.X, screenPos.Y) - center).Magnitude
                        if dist < closestDist and dist <= FOVRadius then
                            closest = head
                            closestDist = dist
                        end
                    end
                end
            end
        end
    end
    return closest
end

    bgCorner.Parent = sliderBackground

    local sliderFill = Instance.new("Frame")
    sliderFill.Name = "SliderFill"
    sliderFill.Size = UDim2.new((defaultValue - minValue) / (maxValue - minValue), 0, 1, 0)
    sliderFill.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    sliderFill.BorderSizePixel = 0
    sliderFill.Parent = sliderBackground

    local fillCorner = Instance.new("UICorner")
    fillCorner.CornerRadius = UDim.new(1, 0)
    fillCorner.Parent = sliderFill

    local sliderButton = Instance.new("Frame")
    sliderButton.Name = "SliderButton"
    sliderButton.Size = UDim2.new(0, 14, 0, 14)
    sliderButton.Position = UDim2.new((defaultValue - minValue) / (maxValue - minValue), -7, 0.5, -7)
    sliderButton.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    sliderButton.BorderSizePixel = 0
    sliderButton.Parent = sliderBackground

    local buttonCorner = Instance.new("UICorner")
    buttonCorner.CornerRadius = UDim.new(1, 0)
    buttonCorner.Parent = sliderButton

    local currentValue = defaultValue
    local dragging = false

    local function updateSlider(input)
        local position = math.clamp((input.Position.X - sliderBackground.AbsolutePosition.X) / sliderBackground.AbsoluteSize.X, 0, 1)
        currentValue = math.floor(minValue + (maxValue - minValue) * position)
        valueLabel.Text = tostring(currentValue)
        TweenService:Create(sliderFill, TweenInfo.new(0.1, Enum.EasingStyle.Quad), 
            {Size = UDim2.new(position, 0, 1, 0)}):Play()
        TweenService:Create(sliderButton, TweenInfo.new(0.1, Enum.EasingStyle.Quad), 
            {Position = UDim2.new(position, -7, 0.5, -7)}):Play()
        if callback then
            callback(currentValue)
        end
    end

    local function startDrag(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            updateSlider(input)
        end
    end

    local function endDrag(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end

    sliderBackground.InputBegan:Connect(startDrag)
    sliderBackground.InputEnded:Connect(endDrag)
    sliderButton.InputBegan:Connect(startDrag)
    sliderButton.InputEnded:Connect(endDrag)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            updateSlider(input)
        end
    end)

    sliderFrame.MouseEnter:Connect(function()
        TweenService:Create(sliderFrame, TweenInfo.new(0.2), 
            {BackgroundColor3 = Color3.fromRGB(30, 30, 30)}):Play()
    end)

    sliderFrame.MouseLeave:Connect(function()
        TweenService:Create(sliderFrame, TweenInfo.new(0.2), 
            {BackgroundColor3 = Color3.fromRGB(25, 25, 25)}):Play()
    end)

    return sliderFrame
end

local function MakeDraggable(gui)
    local dragging = false
    local dragInput, dragStart, startPos
    local function updatePosition(input)
        local delta = input.Position - dragStart
        local newPosition = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X,
                                      startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        local tweenInfo = TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
        TweenService:Create(gui, tweenInfo, {Position = newPosition}):Play()
    end

    gui.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = gui.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    gui.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            updatePosition(input)
        end
    end)
end
